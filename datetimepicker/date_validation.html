                自定义日历框显示内容
                (function($) {

                        days: ["周日", "周一", "周二", "周三", "周四", "周五", "周六", "周日"],
                        daysShort: ["周日", "周一", "周二", "周三", "周四", "周五", "周六", "周日"],
                        daysMin: ["日", "一", "二", "三", "四", "五", "六", "日"],
                        months: ["1月", "2月", "3月", "4月", "5月", "6月", "7月", "8月", "9月", "10月", "11月", "12月"],
                        monthsShort: ["1月", "2月", "3月", "4月", "5月", "6月", "7月", "8月", "9月", "10月", "11月", "12月"],
                        today: "今天",
                        suffix: [],
                        meridiem: ["am", "pm"],
                        weekStart: 1,

                    };
                }(jQuery));   
                自定义显示内容
                $("#form_datetime").datetimepicker({
                    format: "yyyy - MM - dd " + '日' + " - hh:ii",//选择日期时间后，在input中显示的内容配置
                    autoclose: true,//空白处关闭
                    todayBtn: true,//显示'今天'按钮 方便直接选择到今天
                    pickerPosition: "bottom-left",//日历框出现位置
                    language: 'fr'//引用哪个日历框展现形式 en为默认英文 目前没有官方中文 按照上面这个方式配置出中文
                });
